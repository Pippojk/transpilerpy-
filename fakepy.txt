def piramide(n):
    i = 1
    while (i <= n):
        j = 1
        
        while (j <= n - i):
            print(" ", end="")
            j = j + 1
        
        k = 1
        while (k <= i):
            print(k, end="")
            k = k + 1
        
        k = k - 2
        
        while (k >= 1):
            print(k, end="")
            k = k - 1
        
        print("")
        i = i + 1


n = input("Inserisci un numero: ")
piramide(n)
